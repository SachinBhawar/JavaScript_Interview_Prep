push()

Mutates

Adds one or more elements to the end of an array.

Parameters: ...elements (elements to add)
Returns: New length of the array

Use Case:

Dynamically appending data to arrays, building collections

const fruits = ['apple', 'banana'];
fruits.push('orange'); // ['apple', 'banana', 'orange']
fruits.push('kiwi', 'mango'); // Adds multiple items

pop()

Mutates

Removes the last element from an array.

Parameters: None
Returns: Removed element

Use Case:

Stack operations (LIFO), processing items in reverse order

const numbers = [1, 2, 3];
const last = numbers.pop(); // returns 3
// numbers now [1, 2]

shift()

Mutates

Removes the first element from an array.

Parameters: None
Returns: Removed element

Use Case:

Queue operations (FIFO), processing items in order

const colors = ['red', 'green', 'blue'];
const first = colors.shift(); // returns 'red'
// colors now ['green', 'blue']

unshift()

Mutates

Adds one or more elements to the beginning of an array.

Parameters: ...elements (elements to add)
Returns: New length of the array

Use Case:

Prepending data to arrays, priority queues

const nums = [2, 3];
nums.unshift(1); // [1, 2, 3]
nums.unshift(-1, 0); // [-1, 0, 1, 2, 3]

concat()

Pure

Merges two or more arrays into a new array.

Parameters: ...arrays (arrays to concatenate)
Returns: New merged array

Use Case:

Combining arrays without mutation, merging datasets

const arr1 = [1, 2];
const arr2 = [3, 4];
const combined = arr1.concat(arr2);
// [1, 2, 3, 4]

slice()

Pure

Returns a shallow copy of a portion of an array.

Parameters: start (optional), end (optional)
Returns: New array with extracted elements

Use Case:

Extracting subarrays, cloning arrays

const animals = ['ant', 'bison', 'camel'];
animals.slice(1); // ['bison', 'camel']
animals.slice(1, 3); // ['bison', 'camel']
animals.slice(-2); // ['bison', 'camel']

splice()

Mutates

Adds/removes elements at a specific position.

Parameters: start, deleteCount, ...items
Returns: Array of removed elements

Use Case:

Inserting/deleting elements at any position

const months = ['Jan', 'March'];
// Insert at index 1
months.splice(1, 0, 'Feb');
// ['Jan', 'Feb', 'March']

// Replace 1 element at index 2
months.splice(2, 1, 'Mar');
// ['Jan', 'Feb', 'Mar']

map()

Pure

Creates a new array by transforming each element.

Parameters: callback(current, index, array)
Returns: New transformed array

Use Case:

Data transformation, converting data formats

const numbers = [1, 4, 9];
const roots = numbers.map(num => Math.sqrt(num));
// [1, 2, 3]

const names = ['john', 'jane'];
const capitalized = names.map(name =>
  name.charAt(0).toUpperCase() + name.slice(1)
); // ['John', 'Jane']

filter()

Pure

Creates a new array with elements that pass a test.

Parameters: callback(element, index, array)
Returns: New filtered array

Use Case:

Removing invalid data, search filtering

const words = ['spray', 'limit', 'elite'];
const longWords = words.filter(word => word.length > 5);
// ['spray', 'limit']

const nums = [-3, -2, 0, 1, 2];
const positives = nums.filter(n => n > 0);
// [1, 2]

reduce()

Pure

Reduces the array to a single value.

Parameters: callback(acc, current), initialValue
Returns: Accumulated result

Use Case:

Summing values, flattening arrays, data aggregation

const numbers = [1, 2, 3, 4];
const sum = numbers.reduce((acc, curr) => acc + curr, 0);
// 10

const nested = [[1, 2], [3, 4]];
const flat = nested.reduce((acc, curr) =>
  acc.concat(curr), []
); // [1, 2, 3, 4]

forEach()

Pure*

Executes a function for each element.

Parameters: callback(element, index, array)
Returns: undefined

Use Case:

Side effects (logging, DOM updates), iteration without transformation

const items = ['item1', 'item2'];
items.forEach((item, index) => {
  console.log(`${index}: ${item}`);
});
// Logs: "0: item1", "1: item2"

// Modifying external state
let sum = 0;
[1, 2, 3].forEach(num => {
  sum += num;
}); // sum = 6

find()

Pure

Returns the first element that satisfies a condition.

Parameters: callback(element, index, array)
Returns: Found element or undefined

Use Case:

Searching for unique IDs, finding first match

const users = [
  { id: 1, name: 'John' },
  { id: 2, name: 'Jane' }
];
const user = users.find(user => user.id === 2);
// { id: 2, name: 'Jane' }

const nums = [5, 12, 8, 130];
const found = nums.find(num => num > 10);
// 12

findIndex()

Pure

Returns the index of the first element that satisfies a condition.

Parameters: callback(element, index, array)
Returns: Index or -1 if not found

Use Case:

Finding position of an item, then performing operations

const nums = [5, 12, 8, 130];
const index = nums.findIndex(num => num > 10);
// 1

const fruits = ['apple', 'banana', 'mango'];
const idx = fruits.findIndex(fruit => fruit === 'banana');
// 1

includes()

Pure

Checks if an array contains a value.

Parameters: value, fromIndex (optional)
Returns: Boolean

Use Case:

Membership checks, simple existence verification

const pets = ['cat', 'dog', 'bat'];
pets.includes('cat'); // true
pets.includes('bird'); // false

// Case-sensitive
[1, 2, 3].includes(2); // true
[1, 2, 3].includes(4); // false
[1, 2, NaN].includes(NaN); // true

sort()

Mutates

Sorts elements (lexicographically by default).

Parameters: compareFunction(a, b) (optional)
Returns: Sorted array (mutated original)

Use Case:

Ordering data, preparing data for display

const nums = [1, 30, 4, 21];
nums.sort(); // [1, 21, 30, 4] (lex order)
nums.sort((a, b) => a - b); // [1, 4, 21, 30]

const names = ['Alice', 'bob', 'Charlie'];
names.sort((a, b) => a.localeCompare(b));
// ['Alice', 'bob', 'Charlie']
